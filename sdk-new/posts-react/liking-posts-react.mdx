Express appreciation and improve visibility by liking posts:

```typescript
import { usePost } from '@sageprotocol/sdk/react';
import { useWallet } from '@suiet/wallet-kit';

function LikePost({ authorSharedUserId, postId, ownedUserId }) {
  const { likePost } = usePost();
  const { account, signAndExecuteTransaction } = useWallet();

  async function handleLike() {
    const result = await likePost({
      amounts: [0, 0],
      appId: 'your-app-id',
      network: 'testnet',
      authorSharedUserId,
      postId,
      ownedUserId,
      self: account.address
    });

    if (result.ok) {
      await signAndExecuteTransaction({ transaction: result.transaction });
      console.log("Post liked successfully!");
    } else {
      console.error(result.err);
    }
  }

  return <button onClick={handleLike}>Like Post</button>;
}
```
