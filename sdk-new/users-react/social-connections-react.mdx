Sage supports following (one-way) and friendships (two-way) using React hooks.

## Following Users

```typescript
import { useUser } from '@sageprotocol/sdk/react';
import { useWallet } from '@suiet/wallet-kit';

function FollowUser({ ownedUserId, targetSharedUserId }) {
  const { followUser } = useUser();
  const { account, signAndExecuteTransaction } = useWallet();

  async function handleFollow() {
    const result = await followUser({
      amounts: [0, 0],
      followSharedUserId: targetSharedUserId,
      ownedUserId,
      self: account.address,
      appId: 'your-app-id',
      network: 'testnet'
    });

    if (result.ok) {
      await signAndExecuteTransaction({ transaction: result.transaction });
      console.log("Followed user successfully!");
    } else {
      console.error(result.err);
    }
  }

  return <button onClick={handleFollow}>Follow User</button>;
}
```
