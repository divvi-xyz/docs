Bookmark posts, channels, and users:

```typescript
import { useUser } from '@sageprotocol/sdk/react';
import { useWallet } from '@suiet/wallet-kit';

function FavoritePost({ ownedUserId, postId, authorSharedUserId }) {
  const { addFavoritePost } = useUser();
  const { account } = useWallet();

  async function handleFavorite() {
    await addFavoritePost({
      appId: 'your-app-id',
      network: 'testnet',
      postId,
      authorSharedUserId,
      ownedUserId
    });
    console.log("Post favorited!");
  }

  return <button onClick={handleFavorite}>Favorite Post</button>;
}
```
